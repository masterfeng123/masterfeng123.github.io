U03-1 三數排序

題目敘述：

輸入三個正整數a、b、c，將a、b、c從小排到大並輸出。

輸入範例
8 5 6

輸出範例
5 6 8


#include <stdio.h>

int main()
{
	int a, b, c;
	scanf("%d %d %d", &a, &b, &c);
	if (a>b)
	{
		if (b>c)
		{
			printf("%d %d %d", c, b, a);
		}
		else
		{
			if (c>a)
			{
				printf("%d %d %d", b, a, c);
			}
			else
			{
				printf("%d %d %d", b, c, a);
			}
		}
	}
	else//b>a
	{
		if (a>c)
		{
			printf("%d %d %d", c, a, b);
		}
		else
		{
			if (b>c)
			{
				printf("%d %d %d", a, c, b);
			}
			else
			{
				printf("%d %d %d", a, b, c);
			}
		}
			
	}
	return 0;
}
